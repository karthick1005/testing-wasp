🎯 Project Overview

This is a comprehensive Open SaaS application built with the Wasp full-stack framework. The project serves as a complete boilerplate for building modern SaaS applications with authentication, payments, file uploads, admin dashboards, AI integration, and multiple sample applications.

⚡
Wasp Framework
Full-stack React + Node.js with configuration-based development
🔐
Multi-Auth Support
Email/password, Google, GitHub, Discord authentication
💳
Payment Processing
Stripe & LemonSqueezy integration with subscriptions
📁
File Management
AWS S3 integration with secure uploads
🤖
AI Integration
OpenAI GPT integration for AI-powered features
📊
Analytics Dashboard
Google Analytics & Plausible with admin insights

🎯 Key Configuration Sections in main.wasp

app OpenSaaS {
  wasp: { version: "^0.17.0" }
  title: "My Open SaaS App"
  
  // 📧 Email configuration
  emailSender: {
    provider: SendGrid,
    defaultFrom: {
      name: "Open SaaS App",
      email: "me@example.com"
    }
  }
  
  // 🔐 Authentication setup
  auth: {
    userEntity: User,
    methods: {
      email: {
        fromField: { name: "Open SaaS App", email: "me@example.com" },
        emailVerification: { clientRoute: EmailVerificationRoute },
        passwordReset: { clientRoute: PasswordResetRoute },
        userSignupFields: import { getEmailUserFields } from "@src/auth/userSignupFields"
      },
      google: {
        userSignupFields: import { getGoogleUserFields } from "@src/auth/userSignupFields",
        configFn: import { getGoogleAuthConfig } from "@src/auth/userSignupFields"
      }
    },
    onAuthFailedRedirectTo: "/login",
    onAuthSucceededRedirectTo: "/apps"
  }
  
  // 🗄️ Database configuration
  db: {
    seeds: [
      import { seedMockUsers } from "@src/server/scripts/dbSeeds"
    ]
  }
  
  // 📱 Client configuration
  client: {
    rootComponent: import App from "@src/client/App"
  }
}

🗄️ Database Schema Overview

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// 👤 Main user entity with auth and subscription data
model User {
  id                        String          @id @default(uuid())
  email                     String?         @unique
  username                  String?         @unique
  isAdmin                   Boolean         @default(false)
  
  // 💳 Payment fields
  paymentProcessorUserId    String?         @unique
  subscriptionStatus        String?         // 'active', 'cancel_at_period_end', etc.
  subscriptionPlan          String?         // 'hobby', 'pro'
  credits                   Int             @default(3)
  
  // 🔗 Relations
  gptResponses              GptResponse[]
  tasks                     Task[]
  files                     File[]
  contactFormMessages       ContactFormMessage[]
}

// ✅ Task management for AI demo app
model Task {
  id          String   @id @default(uuid())
  description String
  time        String   @default("1")
  isDone      Boolean  @default(false)
  user        User     @relation(fields: [userId], references: [id])
  userId      String
}

// 🤖 AI-generated responses
model GptResponse {
  id        String   @id @default(uuid())
  content   String
  user      User     @relation(fields: [userId], references: [id])
  userId    String
}

// 📁 File uploads with S3 integration
model File {
  id        String   @id @default(uuid())
  name      String
  type      String
  key       String
  uploadUrl String
  user      User     @relation(fields: [userId], references: [id])
  userId    String
}

// 📊 Analytics and statistics
model DailyStats {
  id                        Int             @id @default(autoincrement())
  date                      DateTime        @default(now()) @unique
  totalViews                Int             @default(0)
  userCount                 Int             @default(0)
  paidUserCount             Int             @default(0)
  totalRevenue              Float           @default(0)
  sources                   PageViewSource[]
}

🔧 Hook Usage Examples

// 🌓 Theme Management
const { colorMode, toggleColorMode } = useColorMode();

// ⏱️ Debounced Search
const [searchTerm, setSearchTerm] = useState('');
const debouncedSearchTerm = useDebounce(searchTerm, 300);

useEffect(() => {
  if (debouncedSearchTerm) {
    performSearch(debouncedSearchTerm);
  }
}, [debouncedSearchTerm]);

// 💾 localStorage Integration
const [user, setUser] = useLocalStorage('user', null);
const [preferences, setPreferences] = useLocalStorage('preferences', {
  theme: 'light',
  language: 'en'
});

// 🏠 Conditional Navigation
const isLandingPage = useIsLandingPage();
return (
  <nav>
    {!isLandingPage && <BackButton />}
    <Logo />
  </nav>
);

// User management query example
export const getPaginatedUsers: GetPaginatedUsers<Input, Output> = async (
  { skip, take, orderBy, orderDirection },
  context
) => {
  if (!context.user?.isAdmin) {
    throw new HttpError(403, 'Admin access required');
  }

  const users = await context.entities.User.findMany({
    skip,
    take,
    orderBy: { [orderBy]: orderDirection },
    include: {
      _count: { select: { files: true, tasks: true } }
    }
  });

  const totalCount = await context.entities.User.count();
  
  return { users, totalCount };
};

// Payment operation example  
export const generateCheckoutSession: GenerateCheckoutSession = async (
  { planId },
  context
) => {
  if (!context.user) {
    throw new HttpError(401, 'User not authenticated');
  }

  const session = await createStripeCheckoutSession({
    userId: context.user.id,
    planId,
    successUrl: `${process.env.REACT_APP_CLIENT_URL}/checkout/success`,
    cancelUrl: `${process.env.REACT_APP_CLIENT_URL}/pricing`
  });

  return { sessionUrl: session.url };
};

// File upload operation example
export const createFile: CreateFile = async (
  { fileType, name },
  context
) => {
  if (!context.user) {
    throw new HttpError(401, 'User not authenticated');
  }

  const fileKey = `${context.user.id}/${Date.now()}-${name}`;
  const uploadUrl = await generatePresignedUploadUrl(fileKey, fileType);

  const file = await context.entities.File.create({
    data: {
      name,
      type: fileType,
      key: fileKey,
      uploadUrl,
      userId: context.user.id
    }
  });

  return file;
};

// Type-safe object key extraction

// Date formatting utilities
export function formatDate(date: Date | string, format: 'short' | 'long' = 'short') {
  const d = new Date(date);
  return format === 'short' 
    ? d.toLocaleDateString() 
    : d.toLocaleDateString('en-US', {

        month: 'long', 
        day: 'numeric' 
      });
}

// File size formatting
export function formatFileSize(bytes: number): string {
  const sizes = ['Bytes', 'KB', 'MB', 'GB'];
  if (bytes === 0) return '0 Bytes';
  const i = Math.floor(Math.log(bytes) / Math.log(1024));
  return Math.round(bytes / Math.pow(1024, i) * 100) / 100 + ' ' + sizes[i];
}

// Debounce function
export function debounce<T extends (...args: any[]) => any>(
  func: T,
  wait: number
): (...args: Parameters<T>) => void {
  let timeout: NodeJS.Timeout;
  return (...args: Parameters<T>) => {
    clearTimeout(timeout);
    timeout = setTimeout(() => func.apply(this, args), wait);
  };
}

// Validation utilities
export const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
export const phoneRegex = /^\+?[\d\s\-\(\)]+$/;

export function isValidEmail(email: string): boolean {
  return emailRegex.test(email);
}

export function isValidPhone(phone: string): boolean {
  return phoneRegex.test(phone);
}

// Array utilities
export function chunk<T>(array: T[], size: number): T[][] {
  return Array.from({ length: Math.ceil(array.length / size) }, (_, i) =>
    array.slice(i * size, i * size + size)
  );
}

export function unique<T>(array: T[]): T[] {
  return [...new Set(array)];
}

// Object utilities
export function omit<T extends Record<string, any>, K extends keyof T>(
  obj: T,
  keys: K[]
): Omit<T, K> {
  const result = { ...obj };
  keys.forEach(key => delete result[key]);
  return result;
}

export function pick<T extends Record<string, any>, K extends keyof T>(
  obj: T,
  keys: K[]
): Pick<T, K> {
  return keys.reduce((result, key) => {
    if (key in obj) {
      result[key] = obj[key];
    }
    return result;
  }, {} as Pick<T, K>);
}
