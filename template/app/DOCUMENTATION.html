// User management query example
export const getPaginatedUsers: GetPaginatedUsers<Input, Output> = async (
  { skip, take, orderBy, orderDirection },
  context
) => {
  if (!context.user?.isAdmin) {
    throw new HttpError(403, 'Admin access required');
  }

  const users = await context.entities.User.findMany({
    skip,
    take,
    orderBy: { [orderBy]: orderDirection },
    include: {
      _count: { select: { files: true, tasks: true } }
    }
  });

  const totalCount = await context.entities.User.count();
  
  return { users, totalCount };
};

export function formatFileSize(bytes: number): string {

// Debounce function
export function debounce<T extends (...args: any[]) => any>(
  func: T,
  wait: number
): (...args: Parameters<T>) => void {
  let timeout: NodeJS.Timeout;
  return (...args: Parameters<T>) => {
    clearTimeout(timeout);
    timeout = setTimeout(() => func.apply(this, args), wait);
  };
}

export function chunk<T>(array: T[], size: number): T[][] {
  return Array.from({ length: Math.ceil(array.length / size) }, (_, i) =>

export function omit<T extends Record<string, any>, K extends keyof T>(
  obj: T,
  keys: K[]
): Omit<T, K> {
  const result = { ...obj };
  keys.forEach(key => delete result[key]);
  return result;

export function pick<T extends Record<string, any>, K extends keyof T>(
  obj: T,
  keys: K[]
): Pick<T, K> {
  return keys.reduce((result, key) => {
    if (key in obj) {
      result[key] = obj[key];
    }
    return result;
  }, {} as Pick<T, K>);
